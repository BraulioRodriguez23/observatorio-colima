import React, { FC, useState } from 'react'; // Importa `useState` y `FC` correctamente
import Header from '../components/Horizontalmenu';
import './App.css';

// Importa tus p√°ginas correctamente
import Barometer from '../pages/barometer';
import Home from '../pages/home';
import Indicators from '../pages/indicators';
import Publications from '../pages/publications';
import Inventory from '../pages/inventory';

// Objeto que contiene los componentes de las rutas
const homeComponents: Record<string, React.FC> = {
  Barometer: Barometer,
  Home: Home,
  Indicators: Indicators,
  Publications: Publications,
  Inventory: Inventory, // Corrige el nombre a 'Inventory'
};

const Settings: FC = () => {
  const [active, setActive] = useState<string>('Home');

  const ActiveComponent = homeComponents[active];

  const tabs = {
    Barometer: () => setActive('Barometer'),
    Home: () => setActive('Home'),
    Indicators: () => setActive('Indicators'),
    Publications: () => setActive('Publications'),
    Inventory: () => setActive('Inventory'),
  };

  return (
    <div className="px-4">
      <div className="flex flex-row">
        <Header tabs= {tabs}/> 
      </div>
      {/* Renderiza el componente activo */}
      {ActiveComponent && <ActiveComponent />}
    </div>
  );
};

// Exporta el componente principal `Settings`
export default Settings;
